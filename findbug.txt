  //   }
  // });
  
  compiler.hooks.watchRun.tap('checkWatchingMode', () => {
    isWatching = true;
  });

  compiler.hooks.done.tap('doneCallback', (stats) => {
    if (isWatching && (!stats.hasErrors() || ignoreErrors)) {
      executeOpen();
    }
  });
